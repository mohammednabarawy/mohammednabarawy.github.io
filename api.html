API Code Examples

Python Example 


import invoices_reader

# Initialize the client
client = invoices_reader.Client(api_key='YOUR_API_KEY')

# Process an invoice
invoice = client.process_invoice('invoice.pdf')

# Extract key details
total_amount = invoice.total
vendor_name = invoice.vendor

JavaScript Example 


const InvoicesReader = require('invoices-reader');

// Initialize the client
const client = new InvoicesReader('YOUR_API_KEY');

// Process an invoice
client.processInvoice('invoice.pdf')
    .then(invoice => {
        console.log(invoice.total);
        console.log(invoice.vendor);
    });
